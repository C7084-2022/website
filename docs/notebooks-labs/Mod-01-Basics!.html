Lab 05: Resilient Distributed Datasets

::: {.cell application/vnd.databricks.v1+cell=‘{“title”:““,”showTitle”:false,“inputWidgets”:{},“nuid”:“0efb8166-be63-40c5-9e99-cd0a9a6e16b9”}’ execution_count=0}

%scala
// Lab 05b: Query: Do WordCount using 3 separate statements using following Operations (flatMap, map, reduceByKey)

val rdd1 = sc.textFile("/FileStore/tables/mary.txt")
val rdd2 = rdd1.flatMap(line => line.split(" "))
val rdd3 = rdd2.map(word => (word, 1))
val rdd4 = rdd3.reduceByKey((x,y) => x+y)
rdd4.collect()

:::

::: {.cell application/vnd.databricks.v1+cell=‘{“title”:““,”showTitle”:false,“inputWidgets”:{},“nuid”:“f5c0055f-f90a-400e-9d42-9953a8060c1c”}’ execution_count=0}

%scala
// Lab of5b Query: Pipeline 3 Operations (flatMap, map, reduceByKey) into a single statement

val rdd1 = sc.textFile("/FileStore/tables/mary.txt")

rdd2.collect()
